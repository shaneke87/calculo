def func_cos(x, n):
  cos_value = 0
  for k in range(n):
    cos_value += (-1)**k*(x**(2*k)/math.factorial(2*k))
  return cos_value

from math import *
def fourthfactorial(n):
  if n in (1,2,3,4):
    return n
  else:
    if n == 0:
      return 1
    else:
      return n * fourthfactorial(n-4)
def calculo_n(error):
  x=0.1
  m=2
  cota_error=(fourthfactorial(4*m+1)/(4.**(m+1)*factorial(m+1)))*(x**(m+1))
  while(cota_error >= error):
    m=m+1
    cota_error=(fourthfactorial(4*m+1)/(4.**(m+1)*factorial(m+1)))*(x**(m+1))
  return m
